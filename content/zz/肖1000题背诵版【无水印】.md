<script type="module">
// PDF.js 配置
import * as pdfjsLib from '/pdfjs/build/pdf.mjs';
pdfjsLib.GlobalWorkerOptions.workerSrc = '/pdfjs/build/pdf.worker.mjs';

// 渲染 PDF 函数
async function renderPDF(pdfUrl, containerId) {
  const container = document.getElementById(containerId);
  
  // 清空容器
  container.innerHTML = '<p style="text-align: center; color: #666; padding: 20px;">加载中...</p>';
  
  try {
    // 加载 PDF
    const pdf = await pdfjsLib.getDocument(pdfUrl).promise;
    
    // 清空容器
    container.innerHTML = '';
    
    // 渲染所有页面
    for (let pageNum = 1; pageNum <= pdf.numPages; pageNum++) {
      const page = await pdf.getPage(pageNum);
      
      // 计算合适的缩放比例，适配容器宽度
      const containerWidth = container.clientWidth - 40;
      const viewport = page.getViewport({ scale: 1.0 });
      const scale = Math.min(1.5, containerWidth / viewport.width);
      const scaledViewport = page.getViewport({ scale: scale });
      
      // 创建页面容器
      const pageContainer = document.createElement('div');
      pageContainer.style.marginBottom = '20px';
      pageContainer.style.textAlign = 'center';
      pageContainer.style.maxWidth = '100%';
      pageContainer.style.overflowX = 'auto';
      
      // 创建 canvas 元素
      const canvas = document.createElement('canvas');
      const context = canvas.getContext('2d');
      
      // 提高清晰度，设置物理像素
      const outputScale = window.devicePixelRatio || 1;
      canvas.width = Math.floor(scaledViewport.width * outputScale);
      canvas.height = Math.floor(scaledViewport.height * outputScale);
      canvas.style.width = Math.floor(scaledViewport.width) + 'px';
      canvas.style.height = Math.floor(scaledViewport.height) + 'px';
      
      // 添加 canvas 到页面容器
      pageContainer.appendChild(canvas);
      
      // 添加页面容器到主容器
      container.appendChild(pageContainer);
      
      // 渲染页面
      const transform = outputScale !== 1 ? [outputScale, 0, 0, outputScale, 0, 0] : null;
      const renderContext = {
        canvasContext: context,
        viewport: scaledViewport,
        transform: transform
      };
      
      await page.render(renderContext).promise;
    }
    
    console.log('PDF 全部页面渲染完成');
  } catch (error) {
    console.error('PDF 加载失败:', error);
    container.innerHTML = '<p style="padding: 20px; color: red;">PDF 加载失败，请尝试直接下载</p>';
  }
}

// 页面加载完成后渲染 PDF
document.addEventListener('DOMContentLoaded', function() {
  renderPDF('/pdfs/肖1000题精简背诵版【无水印】.pdf', 'pdf-container');
});

// 监听窗口大小变化，重新渲染 PDF
let resizeTimeout;
window.addEventListener('resize', function() {
  clearTimeout(resizeTimeout);
  resizeTimeout = setTimeout(function() {
    renderPDF('/pdfs/肖1000题精简背诵版【无水印】.pdf', 'pdf-container');
  }, 300);
});
</script>
